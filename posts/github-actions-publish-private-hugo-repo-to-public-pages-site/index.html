# This is a basic workflow to help you get started with Actions

name: Hugo Build & Deploy - Private to Public

# Controls when the action will run. Triggers the workflow on push or pull request
# events but only for the main branch
on:
  push:
    branches: [ main ]

# A workflow run is made up of one or more jobs that can run sequentially or in parallel
jobs:
  # This workflow contains a single job called "build"
  build:
    # The type of runner that the job will run on
    runs-on: ubuntu-latest

    # Steps represent a sequence of tasks that will be executed as part of the job
    steps:
    
    # Check Out Repository:  Based on https://github.com/marketplace/actions/deploy-to-github-pages
    - name: Checkout ğŸ›ï¸
      uses: actions/checkout@v3
      with:
        persist-credentials: false
        submodules: true

    # Setup Hugo in the Ubuntu Runner
    - name: Hugo setup
      uses: peaceiris/actions-hugo@v2
      with:
        # The Hugo version to download (if necessary) and use. Example: 0.58.2
        hugo-version: 'latest'
        # Download (if necessary) and use Hugo extended version. Example: true
        extended: false

    # Runs Hugo to build the Static Site
    - name: Run Hugo
      run: hugo --minify --verbose

# Deploy the Static Site to Public Repo (GitHub Pages)
    - name: Deploy ğŸš€
      uses: JamesIves/github-pages-deploy-action@releases/v4
      with:
        # Setup a deploy ssh key in the public repo, then put the private key as a secret in the private repo 
        ssh-key:  ${{ secrets.DEPLOY_KEY }} 
        repository-name: your-public/repo # Public repo
        branch: main # The branch the action should deploy to.
        folder: public # The folder the action should deploy.
